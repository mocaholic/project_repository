공통 JS적용 가이드 (2015.12.07)

"ui.plugins.js" 의 구현된 기능들 입니다. 실행되는 시점은 하단의 "ui.common.js" 에서 실행 된다.
해당 Plug-in 들의 세부 기능들을 사용하기 위해서는, 하단의 "ui.common.js" 다음에 실행하거나 "$(document).ready()" 시점에 실행하면 된다.

UI.resetCommonJS() : 공통 UI요소 들의 적용된 스크립트를 다시 실행하여 적용, (이미 적용된 대상은 적용되지 않는다.)
UI.resetPageJS() : 페이지의 적용된 UI요소 스크립트를 다시 실행하여 적용, UI.resetCommonJS()도 같이 실행된다. (이미 적용된 대상은 적용되지 않는다.)

<div class="select_area ui-select-box">
   <!-- "ui-input-select" 필수 클래스 (select 자체를 비활성화 하기 위해서는 select에 disabled="disabled") -->

//기본적용 (ui.common.js 에서 실행해 주고 있다. 동적으로 생성한 tab 에 기능을 적용하기 위해서 실행해주면 된다.)
$( target ).tabs({
   eventBlock: false, //true를 설정하면 클릭이벤트 전달을 막는다. (a href=의 동작을 막는다), 기본설정=true

   delayTime: 3000 //autoPlay 지연시간 (1000/1초), 기본설정=4000
});

//현재 선택된 Tab 변경 (탭의 갯수보다 큰 index를 설정하면 동작하지 않는다.)
$( target ).trigger( 'tabs:change_index', 2 );

//이벤트 등록
$( target ).on( 'tabs:change', function ( e, index ) {
   console.log( index ); // 현재 선택된 탭의 index
});

남은시간: 0일 00시 20분 00초

> html

<div class="ui-remain-timer">
   남은시간:
   <!-- "ui-remain-timer-day, ui-remain-timer-hour, ui-remain-timer-minute, ui-remain-timer-second" 필수 클래스 -->
   <!-- 일,시,분,초 중 사용하지 않는 class 는 삭제하면 된다. -->
   <span class="ui-remain-timer-day">0</span><span>일 </span>
   <span class="ui-remain-timer-hour">00</span><span>시 </span>
   <span class="ui-remain-timer-minute">20</span><span>분 </span>
   <span class="ui-remain-timer-second">00</span><span>초 </span>
</div>

> Javascript

//역 타이머 적용하기
$( target ).remainTimer({
   startTime: 0,
   finishTime: 20 * ( 60 * 1000 )//20분 설정
});

//역타이머의 이벤트 받기
$( target ).on( 'remaintimer:timer  remaintimer:complete', function ( e, date ) {
   if ( e.type === 'remaintimer:complete' ) {
      //타이머 종료
   } else {
      console.log( date.day );
      console.log( date.hour );
      console.log( date.minute );
      console.log( date.second );
   }
});

//타이머 시간 재설정
$( target ).trigger( 'remaintimer:reset', {
   startTime: 0,
   finishTime: 1000 * 60 //60초 설정
});

//타이머 정지
$( target ).trigger( 'remaintimer:stop' );

      </div>
      <div class="layer_pop_cont">

         </div>
         <div class="ui-modal-btn-area button_area both_btn">
            <button type="button" class="botton_st_04 purple ui-modal-btn-confirm"></button>
            <button type="button" class="botton_st_04 ui-modal-btn-cancel"></button>
         </div>
      </div>
   </div>
</div>

> Javascript

//Alert
$( target ).click( function (e) {
    //기본 Alert
    UI.alert( 'Alert 띄우기' ).done( function () {
        console.log( '확인 버튼 클릭' );
    });
});

//Confirm
$( target ).click( function (e) {
    //기본 Confirm
    UI.confirm( 'Confirm 띄우기' )
        .done( function () {
            console.log( '확인 버튼 클릭' );
        }).fail( function () {
            console.log( '취소나 "X"버튼 클릭' );
        });

    //각 옵션 설정, 옵션 설정은 alert도 동일
    UI.confirm( 'Confirm 띄우기', {
        textAlignCenter: true, //메세지 내용 가운데 정렬
        title: '팝업의 제목', //팝업 제목 변경
        confirmLabel: '적용', //"확인" 버튼 Label변경
        cancelLabel: '미적용' //"취소" 버튼 Label변경
    });
});
